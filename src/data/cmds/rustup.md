```bash
# Rust のツールチェーンを最新版に更新する
rustup update
# インストール済の Rust のバージョンを列挙する
rustup show
# Rust のコンポーネントを追加する
rustup component add clippy
# Rust のコンポーネントを削除する
rustup component remove clippy
```
